# Spring Market Application

ADD to Application.properties:

- DB_USERNAME = ${DB_USERNAME}

- DB_PASSWORD = ${DB_PASSWORD}

- DB_URL = ${DB_URL}

#For example:

* DB_USERNAME = root

* DB_PASSWORD = super_password

* DB_URL = jdbc:postgresql://localhost:5432/postgres
